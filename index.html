    public void insertLast(int val) {

        current.nextLink = newLink;

    // 5. remove (‡∂Ö‡∂ú‡∂∫‡∂ö‡∑ä ‡∂â‡∑Ä‡∂≠‡∑ä ‡∂ö‡∑í‡∂ª‡∑ì‡∂∏) - (UI ‡∂ë‡∂ö‡∑ö 'remove(value)')
    public Link remove(int value) {

        Link current = first;

        while (current.data != value) {

                current = current.nextLink; // ‡∂ä‡∑Ö‡∂ü ‡∂ë‡∂ö‡∂ß ‡∂∫‡∂±‡∑Ä‡∑è

        }

        // 5c. ‡∂Ö‡∂ú‡∂∫ ‡∑Ñ‡∂∏‡∑î‡∑Ä‡∑î‡∂±‡∑è!

            first = first.nextLink;

            previous.nextLink = current.nextLink;

    }
